create table projeto(
	numero int,
	nome varchar(50),
	localizacao varchar(50));
	
create table empregado(
	rg varchar(12),
	nome varchar(50),
	cpf varchar(11),
	depto varchar(20),
	rg_surpevisor varchar(12),
	sal money,
	dat_init_sal date);
	
create table departamento(
	numero int,
	nome varchar(50),
	rg_gerente varchar(12));
	
create table dependente(
	codigo varchar(10),
	rg_responsavel varchar(12),
	nome_dependente varchar(50),
	nascimento datetime,
	relacao varchar(15),
	sexo varchar(10));
	
create table empregado_projeto(
	codigo varchar(10),
	rg_empregado varchar(12),
	numero_projeto int,
	horas int);
	
create table departamento_projeto(
	codigo varchar(10),
	numero_depto int,
	numero_projeto int);
	
create table historico_salario(
	rg varchar(12),
	dat_ini_sal date,
	dat_fim_sal date,
	salario money);
